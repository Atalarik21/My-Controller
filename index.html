const int mqtt_port = 8883; // Port SSL MQTT const char* mqtt_username = "Attamationhtml"; const char* mqtt_password = "Attamationhtml21"; WiFiClientSecure espClient; PubSubClient client(espClient); const int relay1Pin = D5; // Relay 1 di pin D5 const int relay2Pin = D6; // Relay 2 di pin D6 void setup() { Serial.begin(115200); pinMode(relay1Pin, OUTPUT); pinMode(relay2Pin, OUTPUT); digitalWrite(relay1Pin, HIGH); // Relay OFF di awal (ingat logika terbalik) digitalWrite(relay2Pin, HIGH); WiFi.begin(ssid, password); Serial.println("Connecting to WiFi..."); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi connected"); espClient.setInsecure(); // Untuk SSL tanpa verifikasi sertifikat client.setServer(mqtt_server, mqtt_port); client.setCallback(callback); connectToMqtt(); } void connectToMqtt() { while (!client.connected()) { Serial.print("Connecting to MQTT..."); String clientId = "esp8266Client-" + String(random(0xffff), HEX); if (client.connect(clientId.c_str(), mqtt_username, mqtt_password)) { Serial.println(" connected!"); // Subscribe ke topik kontrol client.subscribe("esp/relay1"); client.subscribe("esp/relay2"); client.subscribe("esp/get_status"); // Publish status aktual relay publishRelayStatus(); } else { Serial.print(" failed, rc="); Serial.print(client.state()); Serial.println(" try again in 5 seconds"); delay(5000); } } } void publishRelayStatus() { // Publish status relay 1 if (digitalRead(relay1Pin) == LOW) { client.publish("esp/status/relay1", "ON"); } else { client.publish("esp/status/relay1", "OFF"); } // Publish status relay 2 if (digitalRead(relay2Pin) == LOW) { client.publish("esp/status/relay2", "ON"); } else { client.publish("esp/status/relay2", "OFF"); } } void callback(char* topic, byte* payload, unsigned int length) { String message; for (unsigned int i = 0; i < length; i++) { message += (char)payload[i]; } Serial.print("Topic: "); Serial.print(topic); Serial.print(" => "); Serial.println(message); if (String(topic) == "esp/relay1") { if (message == "ON") { digitalWrite(relay1Pin, LOW); // Relay ON (logika terbalik) client.publish("esp/status/relay1", "ON"); } else if (message == "OFF") { digitalWrite(relay1Pin, HIGH); // Relay OFF client.publish("esp/status/relay1", "OFF"); } } if (String(topic) == "esp/relay2") { if (message == "ON") { digitalWrite(relay2Pin, LOW); // Relay ON client.publish("esp/status/relay2", "ON"); } else if (message == "OFF") { digitalWrite(relay2Pin, HIGH); // Relay OFF client.publish("esp/status/relay2", "OFF"); } } if (String(topic) == "esp/get_status") { // Jika ada request status dari Web, kirimkan status sekarang publishRelayStatus(); } } void loop() { if (!client.connected()) { connectToMqtt(); } client.loop(); }
